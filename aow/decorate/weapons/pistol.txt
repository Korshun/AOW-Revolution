Actor Handgun : AOWWeapon replaces Pistol 21041
{
	//$Category Weapons
	Tag "Handgun"
	Inventory.PickupMessage "Handgun"
	Obituary "%o was capped by %k's handgun."
	Weapon.PreferredSkin "MarinePistol"

	+WEAPON.WIMPY_WEAPON
	+WEAPON.NOAUTOFIRE
	
	Weapon.AmmoUse 1
	Weapon.AmmoGive 20
	Weapon.AmmoType "NewClipAmmo"
	
	AttackSound "PISTFIRE"
	
	States
	{
	Ready:
		PISG A 1 A_WeaponReady
		Loop
	Deselect:
		NULL A 0 A_Lower
		PISG A 1 A_Lower
		Loop
	Select:
		NULL A 0 A_Raise
		PISG A 1 A_Raise
		Loop
	Fire:
		NULL A 0 A_SpawnItemEx("PistolCasing", 8, 8, ACS_ExecuteWithResult(aow_casingheight, 32), random(1, 3), random(0, 1), random(1, 3), random(45, 60), 128)
		NULL A 0 ACS_ExecuteWithResult(aow_traceroffset, 0, 0)
		NULL A 0 A_FireBullets(2.0, 2.0, -1, 8, "BulletPuff", FBF_USEAMMO | FBF_NORANDOM, 1536)
		PISG B 4 A_GunFlash
		PISG CC 1 A_WeaponReady
		PISG BBB 1 A_WeaponReady
		NULL A 0 A_ReFire
		Goto Ready
	Flash:
		PISF A 1 BRIGHT A_Light2
		PISF A 1 BRIGHT A_Light1
		Goto LightDone
	Spawn:
		PIST A -1
		Stop
	}
}

Actor NewClipAmmo : Ammo
{
	Inventory.Amount 30
	Inventory.MaxAmount 400
}
